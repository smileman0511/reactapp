import React from 'react';
import SearchbarComponent from './components/SearchbarComponent';
import useSearchStore from '../useSearchStore';

const CommunityPage = () => {
  // 1. Zustand에서 검색어 저장 및 페이지 초기화 함수 가져옴
  const { setContent, setPage } = useSearchStore();

  // 2. 검색바에서 '엔터'나 '클릭'이 발생했을 때 실행될 함수 생성.
  const handleSearchSubmit = (value) => {
    console.log("전달받은 검색어:", value);
    
    setContent(value); // 스토어에 검색어 저장
    setPage(1);        // 검색 시 결과 페이지를 1페이지로 초기화
    
    // 여기서 실제 API 호출(axios 등)을 추가로 진행할 수도 있습니다.
  };

  return (
    <div style={{ padding: "20px" }}>
      <h2>커뮤니티 검색</h2>
      
      {/* 3. 공통 컴포넌트를 호출하며 필요한 값을 주입합니다. */}
      <SearchbarComponent 
        placeholder="궁금한 실패 사례를 검색해보세요!" 
        onSubmit={handleSearchSubmit} 
      />
      
      {/* 이후 게시글 리스트 컴포넌트 등이 위치하게 됩니다. */}
    </div>
  );
};

export default CommunityPage;
