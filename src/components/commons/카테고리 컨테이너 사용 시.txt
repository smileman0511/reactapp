
ex) 커뮤니티 페이지에서 카테고리 컨테이너 사용 시

const CommunityPage = () => {
  const { setCategory, setPage } = useSearchStore();

  // 공통 컴포넌트에서 클릭이 일어날 때 실행될 로직을 정의.
  const handleCategoryAction = (category, index) => {
    setCategory(index); // Zustand 데이터 저장
    setPage(1);         // 페이지 초기화
  };

  return (
    <div>
      공통 컴포넌트를 사용하면서 기능을 주입합니다. 
      <CategoryContainer onSelect={handleCategoryAction} />
      이후 검색창이나 게시글 리스트가 오면 됩니다.
    </div>
  );
};

export default CommunityPage;
